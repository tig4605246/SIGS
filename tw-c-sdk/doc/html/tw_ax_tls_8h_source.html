139 static INLINE int TW_SSL_READ(TW_SSL * ssl, char * buf, int len, int32_t timeout) {
140  int ret = -1;
141  uint8_t *read_buf = NULL;
142  DATETIME maxtime, ssl_maxtime, current_time;
143 
144  /* maxtime is used in conjunction with the socket read timeout to quickly read the socket */
145  maxtime = twGetSystemMillisecondCount() + timeout;
146 
147  /* the ssl_read timeout is used if the ssl header is read but the socket timeout is reached */
148  /* before the ssl encrypted data can be read */
149  /* in this case, the read will be allowed to continue until the ssl_maxtime is reached */
150  /* this will eliminate the possibility of timing out a read but returining a success since the */
151  /* initial header was read */
152  /* twcfg.ssl_read_timeout is added to maxtime to ensure that ssl_maxtime>=maxtime */
153  ssl_maxtime = maxtime + twcfg.ssl_read_timeout;
154 
155  if(!ssl || !buf) return -1;
156  /* Anything left from last read? */
157  if (ssl->leftover_cnt && ssl->leftover_ptr) {
158  uint16_t bytes = ssl->leftover_cnt > len ? len : ssl->leftover_cnt;
159  memcpy(buf, ssl->leftover_ptr, bytes);
160  if (ssl->leftover_cnt > len) {
161  ssl->leftover_cnt -= len;
162  ssl->leftover_ptr += bytes;
163  } else {
164  ssl->leftover_cnt = 0;
165  ssl->leftover_ptr = ssl->leftover_buf;
166  }
167  TW_AXTLS_LOG("AXTLS:SSL_READ - Reading from leftover data. Asked for %d, Returned %d", len, bytes);
168  return bytes;
169  }
170 
171  twMutex_Lock(ssl->read_write_mutex);
172  TW_AXTLS_LOG("TW_SSL_READ: About to read");
173  /* Loop until we are unblocked or timeout */
174  while ((ret = ssl_read(ssl->ssl, &read_buf)) == SSL_OK) {
175  /* if the socket timeout has been reached... */
176  current_time = twGetSystemMillisecondCount();
177  if (!twTimeGreaterThan(maxtime, current_time)) {
178  TW_AXTLS_LOG("TW_SSL_READ: socket timeout");
179  /* If the SSL_NEED_RECORD flag is cleared and the ssl_maxtime has not been reached, allow the read to */
180  /* continue until the ssl encrypted data is read or the ssl_maxtime is reached */
181  /* This will prevent the TW_SSL_READ from returning a success (because the header was read) when there */
182  /* should have really been a timeout (because the socket timeout was reached) */
183  /* This will also allow TW_SSL_READ to complete a read if there is actual data on the socket that came in */
184  /* right at the end of the frame timeout */
185  if (!TW_AXTLS_IS_SET_SSL_FLAG && twTimeGreaterThan(ssl_maxtime, current_time)) {
186  TW_AXTLS_LOG("TW_SSL_READ: header waitfor++");
187  /* allow the read to continue for at least one more socket_timeout before checking the ssl_maxtime timeout */
188  maxtime = twGetSystemMillisecondCount() + timeout;
189  } else {
190  /* debugging if/else statements
191  /* if (!twTimeGreaterThan(header_maxtime, twGetSystemMillisecondCount())) {
192  TW_AXTLS_LOG("TW_SSL_READ: header timeout");
193  } else {
194  TW_AXTLS_LOG("TW_SSL_READ: no flag change");
195  } */
196  break;
197  }
198  }
199  }
200  TW_AXTLS_LOG("TW_SSL_READ: read ret: %d", ret);
201  twMutex_Unlock(ssl->read_write_mutex);
202 
203  if (ret > SSL_OK) {
204  uint16_t bytes = ret > len ? len : ret;
205  memcpy(buf, read_buf, bytes);
206  ssl->leftover_cnt = ret > len ? ret - len : 0;
207  TW_AXTLS_LOG("AXTLS:SSL_READ - Asked for %d bytes, Read %d bytes", len, ret);
208  if (ssl->leftover_cnt) {
209  if (ssl->leftover_cnt > ssl->max_leftover) {
210  if (ssl->leftover_buf) TW_FREE(ssl->leftover_buf);
211  TW_AXTLS_LOG("AXTLS:SSL_READ - Allocating %d bytes for leftover buffer", ssl->leftover_cnt);
212  ssl->leftover_buf = (uint8_t *)TW_CALLOC(ssl->leftover_cnt,1);
213  if (!ssl->leftover_buf) {
214  TW_AXTLS_LOG("AXTLS - PANIC! Unable to allocate memory");
215  return -1;
216  }
217  ssl->leftover_ptr = ssl->leftover_buf;
218  ssl->max_leftover = ssl->leftover_cnt;
219  }
220  memcpy(ssl->leftover_buf, read_buf + bytes, ssl->leftover_cnt);
221  TW_AXTLS_LOG("AXTLS:SSL_READ - Copied %d bytes to leftover buffer", ssl->leftover_cnt);
222  }
223  return bytes;
224  } else if (ret == SSL_OK) {
225  /* the WaitFor timed out, meaning we have no more data from the server */
226  TW_AXTLS_LOG("AXTLS:SSL_READ - WaitFor timeout while already begun reading TLS packet");
227 
228  if (ssl->ssl->bm_all_data && strlen((char *)ssl->ssl->bm_all_data)){
229  TW_AXTLS_LOG(TW_ERROR, "AXTLS:SSL_READ: SSL_READ - WaitFor timeout while already begun reading TLS packet. SSL buffer (ssl->bm_all_data) contains data. .");
230  }
231 
232  return TW_READ_TIMEOUT;
233  } else return ret;
234 }

269 static INLINE int TW_VALIDATE_CERT(TW_SSL * ssl, char selfSignedOk) {
270  int res = 1;

272  res = ssl_verify_cert ( ssl->ssl );
